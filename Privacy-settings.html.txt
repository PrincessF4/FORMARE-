function showToast(message) {
const container = document.getElementById("toast-container");
const el = document.createElement("div");
el.className = "toast";
el.innerText = message;

container.appendChild(el);

setTimeout(() => {
el.style.opacity = 0;
el.style.transform = "translateY(-10px)";
setTimeout(() => el.remove(), 300);
}, 2000);
}

/* Utility: Renders a simple switch card */
function renderSettingCard({ title, text, value, onChange }) {

"div",

"div",
{ className: "flex items-center justify-between mb-3" },

"div",

React.createElement("p", { className: "text-gray-600" }, text)
),

React.createElement(
"button",
{
className:
"w-16 h-8 rounded-full transition relative " +
(value ? "bg-[#FF4423]" : "bg-gray-300"),
onClick: onChange
},
React.createElement("span", {
className:
"absolute top-1 h-6 w-6 rounded-full bg-white transition " +
(value ? "right-1" : "left-1")
})
)
)
);
}
